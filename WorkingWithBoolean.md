#Как работать с логическими значениями.
```Java
boolean equals = 4 == 4; //Проверка на равенство, результат: true, т.к. 4 равно 4;
boolean notEquals = 4 != 4; //Проверка на неравенство, результат: false, т.к. 4 равно 4;
boolean stringEquality = "abc" == "bcd"; //Сравнивать можно не только числа, результат: false;
boolean differentTypeEquality = 4.0f == 4;//Типы сравниваемых объектов не обязаны быть одинаковыми, резульат: true;

//Так же существуют операторы для сравнения на больше/больше-равно и меньше/меньше-равно.
boolean more = 6 > 6; //false;
boolean moreEquals = 6 >= 6; //true;
boolean less = 7 < 10; //true;
boolean lessEquals = 7<= 10; //true;

//Для работы с логикой важны логические операции И, ИЛИ и НЕ
boolean and = true && false; // Логическое "И", результат: false;
boolean or = true || false; //Логическое "ИЛИ", результат: true;
boolean not = !true; // Логическое "НЕ", результат: false;

//Логические выражения могут быть довольно большими.
boolean bigEquation = (!(4==5) || true && false);
//сначала вычисляются выражения стоящие внутри: (!(false) || true && false)
//затем вычисляются логические НЕ: (true || true && false)
//следующими идут логические И: (true || false)
//в конце вычисляются ИЛИ: true;
//Результат выражения: true.
```
